        const seededRandom = (seed) => {

            let m = 0x80000000;

            let a = 1103515245;

            let c = 12345;

            let state = seed ? seed : Math.floor(Math.random() * (m - 1));

            return () => {

                state = (a * state + c) % m;

                return state / (m - 1);

            };

        };


        const getTodaySeed = () => {

            const now = new Date();

            return parseInt(`${now.getFullYear()}${String(now.getMonth()+1).padStart(2,'0')}${String(now.getDate()).padStart(2,'0')}`);

        };


        const getPuzzleForDate = (seed) => {

            const rng = seededRandom(seed);

            const availableCategories = [...CATEGORY_BANK].sort(() => 0.5 - rng());

            let selectedCategories = [];

            let seenItems = new Set();

            for (let cat of availableCategories) {

                let hasOverlap = cat.items.some(item => seenItems.has(item));

                if (!hasOverlap && selectedCategories.length < 4) {

                    selectedCategories.push(cat);

                    cat.items.forEach(item => seenItems.add(item));

                }

            }

            if (selectedCategories.length < 4) selectedCategories = availableCategories.slice(0, 4);

            let items = [];

            selectedCategories.forEach(cat => {

                cat.items.forEach(item => {

                    items.push({ text: item, category: cat.category, level: cat.level });

                });

            });

            items.sort(() => 0.5 - rng());

            return { categories: selectedCategories, items: items };

        };

                                    const isSelected = selected.find(s => s.text === item.text);

                                        <button key={idx} onClick={() => handleTileClick(item)} className={`aspect-[4/3] rounded-xl font-black text-[9px] sm:text-[11px] md:text-xs p-2 flex items-center justify-center uppercase transition-all duration-75 shadow-sm border-b-4 ${isSelected ? 'bg-[#033C5A] text-white border-blue-900 translate-y-0.5' : 'bg-[#efefef] text-[#033C5A] border-gray-300 hover:bg-gray-200 active:translate-y-1'}`}>

                                            {item.text}

                                        </button>

                                    );

                                })}

                            </div>

                        )}

                        {gameStatus === 'lost' && (

                            <div className="space-y-2 mb-10 opacity-70 scale-95">

                                {puzzle.categories.filter(c => !solvedGroups.some(s => s.category === c.category)).map((g, i) => (

                                    <div key={i} className={`${levelColors[g.level]} w-full py-4 rounded-xl flex flex-col items-center text-white shadow-sm`}>

                                        <h3 className="font-bold uppercase text-xs tracking-widest">{g.category}</h3>

                                        <p className="text-[10px] font-medium">{g.items.join(', ')}</p>

                                    </div>

                                ))}

                            </div>

                        )}

                        {gameStatus === 'playing' && (

                            <div className="flex flex-col items-center gap-8">

                                <div className="flex items-center gap-3">

                                    <span className="text-[10px] font-black text-gray-400 uppercase tracking-widest">Mistakes Remaining</span>

                                    <div className="flex gap-2">

                                        {[...Array(4)].map((_, i) => (<div key={i} className={`w-3.5 h-3.5 rounded-full transition-all duration-500 shadow-inner ${i < mistakesRemaining ? 'bg-[#033C5A]' : 'bg-gray-200'}`}></div>))}

                                    </div>

                                </div>

                                <div className="flex gap-3 sm:gap-4">

                                    <button onClick={() => setSelected([])} className="px-6 py-2.5 border-2 border-gray-200 rounded-full text-[10px] font-black uppercase text-gray-500 hover:bg-gray-50 transition-colors">Deselect</button>

                                    <button onClick={() => setPuzzle(p => ({...p, items: [...p.items].sort(() => Math.random() - 0.5)}))} className="px-6 py-2.5 border-2 border-gray-200 rounded-full text-[10px] font-black uppercase text-gray-500 hover:bg-gray-50 transition-colors">Shuffle</button>

                                    <button onClick={handleSubmit} disabled={selected.length !== 4} className={`px-10 py-2.5 rounded-full text-[10px] font-black uppercase shadow-xl transition-all ${selected.length === 4 ? 'bg-[#033C5A] text-white hover:scale-105 active:scale-95' : 'bg-gray-100 text-gray-300'}`}>Submit</button>

                                </div>

                            </div>

                        )}

                    </main>

                    <Modal isOpen={showStats} onClose={() => setShowStats(false)} title="Statistics">

                        <div className="grid grid-cols-4 gap-4 text-center mb-8">

                            <div className="flex flex-col"><span className="text-3xl font-black text-[#033C5A]">{stats.played}</span><span className="text-[9px] uppercase font-bold text-gray-400 tracking-tighter">Played</span></div>

                            <div className="flex flex-col"><span className="text-3xl font-black text-[#033C5A]">{stats.played ? Math.round((stats.wins/stats.played)*100) : 0}</span><span className="text-[9px] uppercase font-bold text-gray-400 tracking-tighter">Win %</span></div>

                            <div className="flex flex-col"><span className="text-3xl font-black text-[#033C5A]">{stats.currentStreak}</span><span className="text-[9px] uppercase font-bold text-gray-400 tracking-tighter">Streak</span></div>

                            <div className="flex flex-col"><span className="text-3xl font-black text-[#033C5A]">{stats.maxStreak}</span><span className="text-[9px] uppercase font-bold text-gray-400 tracking-tighter">Max</span></div>

                        </div>

                        {gameStatus !== 'playing' && (

                            <button onClick={() => { 

                                navigator.clipboard.writeText(`Revolutionary Links #${getTodaySeed()%1000}\n${gameStatus==='won'?'Victory! 🦛':'Better luck next semester.'}\nStreak: ${stats.currentStreak}`); 

                                showToast("Result copied!"); 

                            }} className="w-full py-4 bg-[#033C5A] text-white rounded-xl font-black flex items-center justify-center gap-3 shadow-lg hover:bg-blue-900 transition-colors">SHARE <Share2 size={20}/></button>

                        )}

                    </Modal>

                    <Modal isOpen={showHelp} onClose={() => setShowHelp(false)} title="Game Rules">

                        <div className="text-sm text-gray-600 space-y-4 font-medium leading-relaxed">

                            <p>Connect four groups of four items with a common GWU-related theme.</p>

                            <p>Each puzzle has exactly one correct solution. Levels range from <span className="font-bold text-[#AA9868]">Buff (Easy)</span> to <span className="font-bold text-purple-800">Purple (Tricky Wordplay)</span>.</p>

                            <p className="text-xs italic text-gray-400">Avoid making 4 mistakes before finishing!</p>

                        </div>

                    </Modal>

                </div>

            );

        };

        root.render(<App />);

    </script>
